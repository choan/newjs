      def browser(browser, path)
        case(browser.to_sym)
          when :firefox
            FirefoxBrowser.new(path)
          when :safari
            SafariBrowser.new(path)
          when :ie
            IEBrowser.new(path)
          when :konqueror
            KonquerorBrowser.new(path)
          else
            browser
        end
      end
